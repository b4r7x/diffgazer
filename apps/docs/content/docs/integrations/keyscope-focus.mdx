Manage keyboard navigation across multiple UI zones (e.g., sidebar + content + preview). Handles zone transitions via arrow keys and Tab cycling. Requires `KeyboardProvider` — not available in copy mode.

> For the complete API reference, see the [keyscope documentation](https://github.com/b4r7x/keyscope).

---

## useFocusZone

```tsx
import { useFocusZone, useKey } from "keyscope"

type Zone = "sidebar" | "content" | "preview"

function Layout() {
  const { zone, inZone, forZone, setZone } = useFocusZone({
    initial: "sidebar",
    zones: ["sidebar", "content", "preview"] as const,
    tabCycle: ["sidebar", "content", "preview"] as const,
    transitions: ({ zone, key }) => {
      if (zone === "sidebar" && key === "ArrowRight") return "content"
      if (zone === "content" && key === "ArrowLeft") return "sidebar"
      if (zone === "content" && key === "ArrowRight") return "preview"
      return null
    },
  })

  // Zone-specific bindings
  useKey("Enter", () => openItem(), forZone("sidebar"))
  useKey("Enter", () => editContent(), forZone("content"))

  return (
    <div>
      <aside className={inZone("sidebar") ? "ring-1 ring-white" : ""}>
        Sidebar
      </aside>
      <main className={inZone("content") ? "ring-1 ring-white" : ""}>
        Content
      </main>
    </div>
  )
}
```

### Return value

| Property | Type | Description |
|----------|------|-------------|
| `zone` | `T` | Current active zone |
| `setZone` | `(zone: T) => void` | Set zone programmatically |
| `inZone` | `(...zones: T[]) => boolean` | Check if current zone matches |
| `forZone` | `(target: T, extra?) => UseKeyOptions` | Create zone-scoped options for `useKey` |
